window.transferUserData = async () => {
  const { getDocs } = await import('https://www.gstatic.com/firebasejs/10.12.0/firebase-firestore.js');
  const fromUID = prompt('UID des Quell-Accounts (von wem kopieren):');
  if (!fromUID) return;
  const toUID = currentUser.uid;
  if (fromUID === toUID) { toast('Gleicher Account!', 'error'); return; }
  toast('Transfer läuft...', 'success');
  try {
    const batch = writeBatch(db);
    const cols = ['actions', 'habits', 'finance'];
    const userSnap = await getDoc(doc(db, 'users', fromUID));
    if (userSnap.exists()) {
      const data = userSnap.data();
      delete data.email;
      batch.update(doc(db, 'users', toUID), data);
    }
    for (const col of cols) {
      const snap = await getDocs(collection(db, 'users', fromUID, col));
      snap.forEach(d => {
        batch.set(doc(collection(db, 'users', toUID, col)), d.data());
      });
    }
    const logsSnap = await getDocs(collection(db, 'users', fromUID, 'dailyLogs'));
    logsSnap.forEach(d => {
      batch.set(doc(db, 'users', toUID, 'dailyLogs', d.id), d.data());
    });
    await batch.commit();
    toast('✅ Alle Daten übertragen!', 'success');
  } catch(e) {
    toast('Fehler: ' + e.message, 'error');
  }
};
